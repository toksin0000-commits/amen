28
فرمانبرداری کی برکتیں
1 اگر تُم یَاہوِہ اَپنے خُدا کی پُوری طور سے اِطاعت کروگے اَورجو اَحکام میں آج تُمہیں دے رہا ہُوں اُن پر احتیاط سے عَمل کروگے تو یَاہوِہ تمہارے خُدا تُمہیں دُنیا کی سَب قوموں سے زِیادہ سرفراز کریں گے۔ 2 اَور اگر تُم واقعی یَاہوِہ اَپنے خُدا کی اِطاعت کروگے تو یہ ساری برکتیں تُم پر نازل ہوں گی اَور تمہارے ساتھ رہیں گی:
3 تُم شہر اَور کھیت میں کام کرتے ہُوئے مُبارک ہوگے۔
4 تمہاری اَولاد مُبارک ہوگی، تمہاری زمین کی پیداوار اَور تمہارے چَوپایوں کے بچّے یعنی گائے بَیل کے بچھڑے اَور بھیڑ بکریوں کے برّے مُبارک ہوں گے۔
5 تمہارا ٹوکرا اَور تمہاری کٹھوتی دونوں مُبارک ہوں گے۔
6 اَور تُم گھر کے اَندر آتے وقت اَور باہر جاتے وقت بھی مُبارک ہوگے۔
7 یَاہوِہ تمہارے دُشمنوں کو جو تمہارے خِلاف سَر اُٹھائیں گے تمہارے سامنے شِکست دِلائیں گے۔ وہ ایک سمت سے آکر تُم پر حملہ کریں گے لیکن سات سمتوں کی طرف تمہارے سامنے سے بھاگ جایٔیں گے۔
8 یَاہوِہ تمہارے اناج کے ذخیروں میں اَور ہر اُس چیز پر جِس میں تُم ہاتھ لگاؤگے برکت نازل کریں گے۔ اَور یَاہوِہ تمہارے خُدا تُمہیں اُس مُلک میں جسے وہ تُمہیں عنایت کر رہے ہیں برکت بخشیں گے۔
9 اگر تُم یَاہوِہ اَپنے خُدا کے اَحکام کو مانوگے اَور اُن کی راہوں پر چلوگے تو یَاہوِہ اَپنی اُس قَسم کے مُطابق جو اُنہُوں نے تُم سے کھائی تھی تُمہیں اَپنی پاک قوم بنا کر قائِم کریں گے۔ 10 تَب دُنیا کی سَب قومیں دیکھیں گی کہ تُم یَاہوِہ کی قوم ہو اَور وہ تُم سے ڈریں گی۔ 11 جِس مُلک کو تُمہیں دینے کی قَسم یَاہوِہ نے تمہارے آباؤاَجداد سے کھائی تھی اُس میں یَاہوِہ تمہاری اَولاد کو، تمہارے مویشیوں کے بچّوں کو اَور تمہاری زمین کی فصل کو خُوب بڑھا کر تُمہیں کثرت سے ترقّی بخشیں گے۔
12 یَاہوِہ آسمان کو جو اُن کا عظیم مخزن ہے تمہارے لیٔے کھول دیں گے تاکہ تمہارے مُلک میں وقت پر مینہ برسے اَور تمہارے ہاتھ کے سَب کاموں کو جِن میں تُم ہاتھ لگاؤ برکت دیں گے؛ اَور تُم بہت سِی قوموں کو قرض دوگے لیکن آپ کسی کے مقروض نہ ہوگے۔ 13 اَور یَاہوِہ تُمہیں دُم نہیں بَلکہ سَر ٹھہرائیں گے؛ بشرطیکہ تُم یَاہوِہ اَپنے خُدا کے اُن اَحکام کو جنہیں آج مَیں تُمہیں دے رہا ہُوں، نہایت احتیاط سے اُن پر غور کرو اَور عَمل کرو، تو تُم ہمیشہ سرفراز رہوگے اَور کبھی نیچے نہ ہوگے۔ 14 اَورجو اَحکام میں آج تُمہیں دے رہا ہُوں اُن میں سے کسی سے تُم داہنے یا بائیں نہ مُڑنا اَور نہ ہی غَیر معبُودوں کی پیروی اَور خدمت کرنا۔
نافرمانی کی لعنتیں
15 اگر تُم یَاہوِہ اَپنے خُدا کی اِطاعت نہ کروگے اَور اُن کے سَب اَحکام اَور قوانین پرجو آج مَیں تُمہیں دے رہا ہُوں نہایت احتیاط سے عَمل نہ کروگے، تو تُم پر یہ سَب لعنتیں نازل ہوں گی اَور تمہارا یہ حال ہوگا:
16 تُم شہر میں اَور کھیت میں کام کرتے وقت بھی لعنتی ہوگے۔
17 تمہارا ٹوکرا اَور تمہاری کٹھوتی دونوں لعنتی ہوں گے۔
18 تمہاری اَولاد لعنتی ہوگی اَور تمہاری زمین کی پیداوار اَور تمہارے گائے، بَیل کے بچھڑے اَور تمہاری بھیڑ بکریوں کے برّے لعنتی ہوں گے۔
19 تُم گھر کے اَندر آتے وقت اَور باہر جاتے وقت بھی لعنتی ہوگے۔
20 تُم نے یَاہوِہ کو ترک کرکے جو بدکاری کی ہے اُس کے سبب سے یَاہوِہ اُن سَب کاموں پر جِن کو تُم ہاتھ لگاؤگے، لعنتیں، اُلجھن اَور ملامت بھیجتے رہیں گے، جَب تک کہ تُم ہلاک نہ ہو جاؤ اَور جلدی برباد نہ ہو جاؤ۔ 21 یَاہوِہ اُس وقت تک تُم پر وبَائیں نازل کرتے رہیں گے، جَب تک کہ وہ تُمہیں اُس مُلک سے جِس پر قبضہ کرنے کے لیٔے تُم اُس میں داخل ہو رہے ہو، نِیست و نابود نہ کر دیں۔ 22 یَاہوِہ تُم پر تپِ دِق، بُخار اَور سوزِش، شدید تپش اَور خشک سالی کی مار اَور پالا اَور پھپھُوندی نازل کریں گے جو تمہارے مرنے تک پیچھے لگی رہیں گی۔ 23 اَور تمہارے سَر کے اُوپر کا آسمان کانسے کا اَور تمہارے نیچے کی زمین لوہے کی ہو جایٔےگی۔ 24 اَور یَاہوِہ تمہارے مُلک میں دُھول اَور ریت کی بارش کو بھیجیں گے؛ اَور جَب تک تُم نِیست و نابود نہیں ہو جاتے یہ آسمان سے برستی ہی رہیں گی۔
25 یَاہوِہ تُمہیں تمہارے دُشمنوں سے شِکست دِلائیں گے۔ تُم اُن پر ایک سمت سے حملہ کروگے لیکن سات سمتوں کی جانِب اُن کے سامنے سے بھاگ جاؤگے؛ اَور دُنیا کی تمام سلطنتوں کے لوگ تمہاری حالت دیکھ کر دہشت کھایٔیں گے۔ 26 تمہاری لاشیں ہَوا کے تمام پرندوں اَور زمین کے جنگلی جانوروں کی خُوراک ہوں گی اَور کویٔی نہ ہوگا جو اُنہیں ڈرا کر بھگا سکے۔ 27 یَاہوِہ تُمہیں مِصر کے پھوڑوں، گلٹیوں، ناسور اَور خارِش میں اِس طرح مُبتلا کر دیں گے کہ تُم کبھی شفایاب نہ ہو پاؤگے۔ 28 یَاہوِہ تُمہیں پاگل پن، نابینائی اَور دِل کی گھبراہٹ میں بھی مُبتلا کر دیں گے۔ 29 اَور جَیسا نابینا اَندھیرے میں ٹٹولتا ہے وَیسے ہی تُم دوپہر کے وقت ٹٹولتے پھروگے۔ اَور تُم جو کچھ کروگے اُس میں ناکام رہوگے۔ دِن بہ دِن تُم پر ظُلم ڈھائے جایٔیں گے اَور تُم لُٹتے ہی رہوگے اَور تُمہیں بچانے والا کویٔی نہ ہوگا۔
30 جِس عورت سے تمہاری منگنی ہوگی، کویٔی دُوسرا شخص اُسے اَپنا بنا لے گا اَور اُس سے ہم بِستر ہوگا۔ تُم گھر بناؤگے لیکن اُس میں بسنے نہ پاؤگے۔ تُم انگوری باغ لگاؤگے لیکن اُس کے پھل کا لُطف نہ اُٹھا پاؤگے۔ 31 تمہارا بَیل تمہاری آنکھوں کے سامنے ذبح کیا جائے گا لیکن تُم اُس کا گوشت کھانے نہ پاؤگے۔ تمہارا گدھا تُم سے زبردستی چھین لیا جائے گا اَور لَوٹایا نہ جائے گا۔ تمہاری بھیڑیں تمہارے دُشمنوں کو دے دی جایٔیں گی اَور کویٔی اُنہیں بچا نہ پایٔےگا۔ 32 تمہارے بیٹے اَور بیٹیاں دُوسری قوم کو دئیے جایٔیں گے اَور تمہاری آنکھیں دِن بہ دِن اُن کا اِنتظار کرتے کرتے تھک جایٔیں گی مگر تمہارا کچھ بس نہ چلے گا۔ 33 تمہاری زمین کی پیداوار اَور تمہاری محنت و مشقّت کی کمائی ایک اَیسی قوم کھائے گی جِس سے تُم واقف نہیں ہو اَور عمر بھر شدید ظُلم ہی تمہارے پَلّے پڑےگا۔ 34 جو منظر تُم اَپنی آنکھوں سے دیکھوگے وہ تُمہیں پاگل بنا دیں گے۔ 35 یَاہوِہ تمہارے گھٹنوں اَور ٹانگوں میں اَیسے دردناک پھوڑے پیدا کریں گے جو لاعلاج ہوں گے اَورجو تمہارے پاؤں کے تلووں سے لے کر سَر کی چاند تک پھیلے ہوں گے۔
36 یَاہوِہ تُمہیں اُس بادشاہ سمیت جسے تُم اَپنے اُوپر مُقرّر کروگے ایک اَیسی قوم میں دفع کریں گے جِس سے تُم اَور تمہارے آباؤاَجداد ناواقِف ہوں گے۔ وہاں تُم غَیر معبُودوں کی عبادت کروگے جو لکڑی اَور پتّھر کے بنے ہویٔے ہیں۔ 37 تَب تُم اُن سَب قوموں کے درمیان جہاں یَاہوِہ تُمہیں دفع کریں گے، تُم لوگوں کے لیٔے ایک دہشت کی شَے، ضرب المثل اَور ایک مذاق کا باعث بنوگے۔
38 تُم کھیت میں کافی بیج ڈالوگے لیکن بہت ہی کم فصل کاٹوگے کیونکہ ٹِڈّیاں اُسے کھا جایٔیں گی۔ 39 تُم انگوری باغ لگاؤگے اَور اُن میں کاشتکاری کروگے لیکن تُم اُس کا انگوری شِیرہ نہ پیوگے اَور نہ انگور جمع کر پاؤگے کیونکہ اُنہیں کیڑے کھا جایٔیں گے۔ 40 تمہارے سارے مُلک میں زَیتُون کے درخت لگے ہوں گے، لیکن تُم زَیتُون کا تیل اِستعمال نہ کر پاؤگے کیونکہ زَیتُون کے پھل جھڑ جایٔیں گے۔ 41 تمہارے ہاں بیٹے اَور بیٹیاں پیدا ہوں گی لیکن تُم اُنہیں اَپنے پاس رکھ نہ پاؤگے کیونکہ وہ اسیر ہوکر چلے جایٔیں گے۔ 42 تمہارے سَب درختوں اَور تمہاری زمین کی ساری فصلوں پر ٹِڈّیوں کے غول قبضہ کر لیں گے۔
43 تمہارے درمیان رہنے والا پردیسی تُم سے زِیادہ ترقّی کرتا اَور سرفراز ہوتا جائے گا لیکن تُم روز بروز ناکامیاب اَور برباد ہوتے چلے جاؤگے۔ 44 وہ پردیسی تُمہیں قرض دیں گے لیکن تُم اُنہیں قرض نہ دے سکوگے۔ اَور وہ سَر ہوں گے اَور تُم دُم ٹھہروگے۔
45 یہ سَب لعنتیں تُم پر نازل ہوں گی۔ وہ تمہارا تعاقب کریں گی اَور تُمہیں دبوچ لیں گی تاکہ تُم فنا ہو جاؤ، کیونکہ تُم نے یَاہوِہ اَپنے خُدا کی اِطاعت نہ کی اَور اُن اَحکام اَور قوانین پر عَمل نہ کیا جو اُنہُوں نے تُمہیں دئیے تھے۔ 46 یہ لعنتیں تمہارے اَور تمہاری نَسل کے لیٔے ہمیشہ تک نِشان اَور معجزہ بَن کر رہیں گے۔ 47 کیونکہ تُم نے خُوشحالی کے ایّام میں یَاہوِہ اَپنے خُدا کی خُوشی اَور شادمانی کے ساتھ خدمت نہ کی، 48 اِس لیٔے تُم بھُوکے اَور پیاسے اَور عُریانی اَور غربت کی حالت میں، اَپنے دُشمنوں کی خدمت کروگے جنہیں یَاہوِہ تمہارے خِلاف بھیجیں گے اَور وہ تمہاری گردن پر لوہے کا جُوا رکھّے رہیں گے، جَب تک تُمہیں نِیست و نابود نہ کر دیں۔
49 جِس طرح عُقاب جھپٹ کر نیچے آتا ہے اُسی طرح یَاہوِہ دُور دراز سے بَلکہ زمین کے اِنتہا سے ایک اَیسی قوم کو تمہارے اُوپر چڑھا لائیں گے جِس کی زبان کو تُم سمجھ نہ پاؤگے۔ 50 وہ ایک غضبناک اَور پتّھر دِل قوم ہوگی جو نہ ضعیفوں کا اِحترام کرےگی اَور نہ جَوانوں پر رحم کھائے گی۔ 51 اِس کے علاوہ وہ قوم تَب تک تمہارے مویشیوں کے بچّے اَور تمہاری زمین کی فصل کو کھاتے رہیں گے جَب تک کہ تُم تباہ نہ ہو جاؤ، یعنی وہ تمہارے لیٔے نہ اناج، نہ نئے انگوری شِیرے یا زَیتُون کا تیل نہ گائے، بَیلوں کے بچھڑے یا تمہاری بھیڑ بکریوں کے برّے چھوڑیں گے، جَب تک کہ تُم فنا نہ ہو جاؤ۔ 52 وہ تمہارے مُلک کے تمام شہروں کا محاصرہ کئے رہیں گے؛ جَب تک کہ تمہاری اُونچی اُونچی اَور مضبُوط فصیلیں جِن پر تُمہیں بھروسا ہے گِر نہ جایٔیں۔ وہ تمہارے مُلک کے سَب شہروں کا محاصرہ کر لیں گے، جنہیں یَاہوِہ تمہارے خُدا تُمہیں دے رہے ہیں۔
53 محاصرہ کے دَوران اَپنے دُشمنوں سے پہُنچی ہُوئی اَذیّت سے تنگ آکر تُم اَپنی اَولاد یعنی اَپنے بیٹوں اَور بیٹیوں کا گوشت کھاؤگے، جنہیں یَاہوِہ تمہارے خُدا نے تُمہیں دیا ہے۔ 54 اُس وقت تمہارے درمیان کا سَب سے نرم دِل اَور حسّاس آدمی بھی اَپنے بھایٔی، اَپنی چہیتی بیوی اَور اَپنے باقی بچے بچّوں پر رحم نہ کھائے گا، 55 اَور وہ اُن میں سے کسی کو بھی اَپنے بچّوں کے گوشت میں سے جو وہ خُود کھا رہا ہوگا کچھ نہ دے گا کیونکہ تمہارے تمام شہروں کے محاصرہ کے دَوران تمہارے دُشمنوں کی نازل کی ہُوئی مُصیبتوں کی وجہ سے اُس کے پاس صِرف وُہی چیز باقی بچی ہوگی۔ 56 تمہارے درمیان وہ عورت بھی جو یہاں تک نرم دِل اَور حسّاس ہو کہ اَپنی نزاکت کی وجہ سے اَپنے پاؤں کا تلوا بھی زمین سے لگانے کی جُرأت نہ کرتی ہو، تو بھی وہ اَپنے محبُوب خَاوند اَور اَپنے بیٹے اَور بیٹی کے خِلاف خُود غرض ہو جایٔےگی۔ 57 وہ اَپنے رِحم سے نکلی ہُوئی نَوزائیدہ بچّہ کو اَور اَپنے فرزند کو چھُپ چھُپ کر کھانا چاہے گی کیونکہ دُشمنوں کی طرف سے اَذیّت ناک محاصرہ کے دَوران تمام اشیائے خُوردنی کی قِلّت ہو جائے گی۔
58 اگر تُم اِس آئین کی تمام باتوں پرجو اِس کِتاب میں لکھی گئی ہیں نہایت احتیاط سے عَمل نہ کروگے اَور یَاہوِہ اَپنے خُدا کے جلالی اَور مُہیب نام کا خوف نہ مانوگے، 59 تَب یَاہوِہ تُم پر اَور تمہاری اَولاد پر خوفناک بَلائیں نازل کریں گے جو سخت اَور لمبے عرصہ کی آفتیں، لمبی اَور شدید عالمی وبَائیں ہوں گی۔ 60 وہ تُمہیں مِصر کی اُن سَب بیماریوں میں مبتلا کر دیں گے جِن سے تُم ڈرتے تھے اَور وہ تُمہیں لگی رہیں گی۔ 61 اَور جَب تک تُم فنا نہ ہو جاؤ، یَاہوِہ تُم پر اَور بھی طرح طرح کی ناتوانیاں اَور آفتیں نازل کریں گے جِن کا ذِکر اِس کِتاب تورہ میں نہیں کیا گیا ہے۔ 62 تُم جو تعداد میں آسمان کے تاروں کی مانند تھے شُمار میں چند ہی رہ جاؤگے کیونکہ تُم نے یَاہوِہ اَپنے خُدا کی اِطاعت نہ کی۔ 63 جَیسے یَاہوِہ تُمہیں خُوشحال کرکے اَور تمہاری تعداد میں اِضافہ کرکے خُوش ہویٔے تھے وَیسے ہی اَب تُمہیں تباہ و برباد کرکے اُنہیں خُوشی ہوگی۔ تُم اُس مُلک سے بے دخل کر دئیے جاؤگے جِس پر قبضہ کرنے کے لیٔے جا رہے ہو۔
64 تَب یَاہوِہ تُمہیں زمین کے ایک سِرے سے دُوسرے سِرے تک سَب قوموں میں پراگندہ کر دیں گے، جہاں تُم لکڑی اَور پتّھر کے بنے ہویٔے معبُودوں کی عبادت کروگے جِن سے نہ تُم اَور نہ تمہارے آباؤاَجداد ہی واقف تھے۔ 65 اُن قوموں میں تُمہیں نہ تو راحت ملے گی اَور نہ ہی تمہارے پاؤں کو آرام کی جگہ نصیب ہوگی۔ وہاں یَاہوِہ کی طرف سے تُمہیں فِکرمند دِل، دھُندھلی آنکھیں اَور غمگین رُوح دی جائے گی۔ 66 تمہاری جان مُستقِل شک میں مُبتلا رہے گی۔ تُم رات دِن خوفزدہ رہوگے اَور تُمہیں اَپنی زندگی کا کویٔی بھروسا نہ ہوگا۔ 67 اُس خوف کے باعث جو تمہارے دِلوں میں سمایا ہوگا اَور اُن نظاروں کے سبب سے جنہیں تمہاری آنکھیں دیکھیں گی تُم صُبح کو کہو گے، ”کاش کہ یہ شام ہوتی!“ اَور شام کو کہو گے، ”کاش یہ صُبح ہوتی!“ 68 اَور یَاہوِہ تُمہیں کشتیوں پر سوار کرکے اُس راستہ سے مِصر لَوٹائے گا، جِس کے بارے میں، مَیں نے تُم سے فرمایا تھا کہ اَب تُم اِسے پھر کبھی نہ دیکھوگے۔ تَب تُم وہاں خُود کو اَپنے دُشمنوں کے سامنے غُلاموں اَور لونڈیوں کی طرح بیچنے کے لیٔے پیش کروگے لیکن تمہارا کویٔی خریدار نہ ہوگا۔
